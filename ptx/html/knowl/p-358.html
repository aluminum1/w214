\begin{equation} k_1 (-1, \, 2) + k_2 (1, \, 1) + k_3 (2, \, -1) = (0, \, 0)\tag{2.2.3} \end{equation}

\begin{equation} -k_1 + k_2 + 2k_3 = 0, \quad 2k_1 + k_2 - k_3 = 0\tag{2.2.4} \end{equation}

\begin{equation} k_1 = t, \, k_2 = -t, \, k_3 = t, t \in \mathbb{R}\text{.}\tag{2.2.5} \end{equation}
