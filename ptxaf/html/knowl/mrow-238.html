\begin{align*} ((\mat{A}\mat{B})\mat{C})_{ij} \amp = \sum_{p=1}^r (\mat{A} \mat{B})_{ip} \mat{C}_{pj}\\ \amp = \sum_{p=1}^r \left(\sum_{q=1}^k \mat{A}_{iq} \mat{B}_{qp} \right) \mat{C}_{pj}\\ \amp = \sum_{p,q} \mat{A}_{iq} \mat{B}_{qp} \mat{C}_{pj}\\ \amp = \sum_{q=1}^k \mat{A}_{iq} \left( \sum_{p=1}^r \mat{B}_{qp} \mat{C}_{pj}\right)\\ \amp = \sum_{q=1}^k \mat{A}_{iq} (\mat{B} \mat{C} )_{qj}\\ \amp = (\mat{A} (\mat{B} \mat{C}))_{ij} \end{align*}
